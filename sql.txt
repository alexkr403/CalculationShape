CREATE TABLE ARTICLE 
(
	ID INTEGER NOT NULL,
	NAME VARCHAR(20) NOT NULL
	CONSTRAINT [PK_ARTICLE] PRIMARY KEY CLUSTERED 
	(
		ID ASC
	)
)

CREATE TABLE TAG
(
	ID INTEGER NOT NULL,
	NAME VARCHAR(20) NOT NULL
	CONSTRAINT [PK_TAG] PRIMARY KEY CLUSTERED 
	(
		ID ASC
	)
)

CREATE TABLE ARTICLE_TAG 
(
	ID_ARTICLE INTEGER NOT NULL,
	ID_TAG INTEGER NOT NULL
	CONSTRAINT [PK_ARTICLE_TAG] PRIMARY KEY CLUSTERED 
	(
		ID_ARTICLE, ID_TAG ASC
	)
)
ALTER TABLE ARTICLE_TAG ADD CONSTRAINT FK_ARTICLE_TAG FOREIGN KEY (ID_TAG) REFERENCES TAG(ID);
ALTER TABLE ARTICLE_TAG ADD CONSTRAINT FK_ARTICLE_TAG1 FOREIGN KEY (ID_ARTICLE) REFERENCES ARTICLE(ID);

SELECT 
	a.NAME as ARTICLE_NAME
    ,t.NAME as TAG
FROM ARTICLE a
LEFT JOIN ARTICLE_TAG arta ON arta.ID_ARTICLE = a.ID
LEFT JOIN TAG t ON t.ID = arta.ID_TAG
